2 * 0 / 1024 - 1		= -1;

2 * 2 / 1024 - 1		= -0,996
2 * 3 / 1024 - 1		= -0,994
2 * 4 / 1024 - 1		= -0,992

2 * 50 / 1024 - 1		= -0,902

2 * 256 / 1024 - 1		= -0,5			/x -> kwart resolutie;
2 * 512 / 1024 - 1		= 0				/x -> helft resolutie;
2 * 768 / 1024 - 1		= 0,5			/x -> 3/4e resolutie;
2 * 1024 / 1024 - 1		= 1				/x -> volledige resolutie;
